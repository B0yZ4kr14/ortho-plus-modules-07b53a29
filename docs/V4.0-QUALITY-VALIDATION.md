```yaml
# .github/workflows/quality.yml
name: Quality Check
on: [push, pull_request]
jobs:
  validate:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v3
      - run: npm install
      - run: npm run validate:quality
      - run: npm run test
      - run: npm run build
```
